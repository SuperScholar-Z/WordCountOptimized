{
	int temp;
	for(int i =8;i>0;i--)
	{
		for(int j =0;j<i;j++)
		{
			if(a[j]%10>a[j+1]%10)
			{
				temp = a[j];
				a[j] = a[j+1];
				a[j+1] = temp;
			}
		}
	}
	for(int i = 8;i>0;i++)
	{
		for(int j = 0;j<i;j++)
		{
			if((a[j]%100-a[j]%10)/10>(a[j+1]%100-a[j+1]%10)/10)
			{
				temp = a[j];
				a[j] = a[j+1];
				a[j+1] = temp;
			}
		}
	}
	for(int i = 8;i>0;i++)
	{
		for(int j = 0;j<i;j++)
		{
			if(a[j]/100>a[j+1]/100)
			{
				temp = a[j];
				a[j] = a[j+1];
				a[j+1] = temp;
			}
		}
	}
}
